\section{Introduction and Motivation}
\begin{frame}{Introduction}
\begin{itemize}
    \item Recent advancements in language models (e.g., ChatGPT) have sparked widespread concerns about labor market disruption.
    \item This study investigates the initial impact of AI on labor demand and productivity by examining online job postings across the EU from 2021--2024.
    \item Main approach: difference-in-differences (DiD) design with ChatGPT’s release as the event date, and multiple AI-exposure measures.
\end{itemize}
\end{frame}

\begin{frame}{Key Findings}
\begin{itemize}
    \item Occupations with higher AI exposure experienced a \textbf{13--18\% larger decrease} in job postings relative to less exposed ones.
    \item Within occupations, skills most similar to AI capabilities saw a \textbf{10\% reduction} in mention frequency.
    \item No significant impact on capital or labor productivity was detected at the industry level so far.
\end{itemize}
\end{frame}

\section{Empirical Strategy}
\begin{frame}{Conceptual Framework}
\begin{itemize}
    \item Standard task-based model where tasks can be performed by labor or AI.
    \item AI exposure of an occupation: fraction of its tasks likely to be automated by language-model-based technologies.
    \item DiD approach (continuous treatment) with ChatGPT release (Nov 2022) as the event.
\end{itemize}
\end{frame}

\begin{frame}{Difference-in-Differences Setup}
\begin{equation*}
    \Delta \log \overline{y}_{i,c} = \delta_0 + \xi_c + \delta \cdot X^{AI}_i + \varepsilon_{i,c},
\end{equation*}
\begin{itemize}
    \item $\overline{y}_{i,c}$ = avg. number of online job ads for occupation $i$ in country $c$, before vs. after the event date.
    \item $X^{AI}_i$: AI exposure for occupation $i$.
    \item $\xi_c$: country fixed effects.
    \item Also tested event-study, decile-based exposure, skill-level analyses, etc.
\end{itemize}
\end{frame}

\section{Conclusion}
\begin{frame}{Conclusion}
\begin{itemize}
    \item Early evidence (2021--2024) suggests \textbf{occupations highly exposed to AI experienced a distinctly larger drop} in online job postings.
    \item \textbf{Task-level changes:} skill mentions that overlap with AI capabilities declined.
    \item \textbf{No measurable productivity boost} so far at the industry level.
    \item Overall, \emph{AI-induced displacement} is nontrivial, but the total effect is still relatively modest in scale.
\end{itemize}
\end{frame}

\begin{frame}{Limitations and Future Work}
\begin{itemize}
    \item \textbf{Data coverage and “ghost postings”} in online adverts can bias results.
    \item Potential \textbf{pre-trends} and endogeneity issues in difference-in-differences.
    \item Future research: 
    \begin{itemize}
        \item More granular firm-level data
        \item Longer post-treatment window
        \item Complementary vs. displacement tasks
    \end{itemize}
\end{itemize}
\end{frame}
